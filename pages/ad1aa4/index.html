# if

number = 23

guess = int(input('Enter an inter:'))


if guess > number:
    print('输入大于{}, {}'.format(number, guess))
elif guess == number:
    print('输入等于{}, {}'.format(number, guess))
else:
    print('输入小于{}, {}'.format(number, guess))

# while

number = 23
running = True
while running:
    guess = int(input('Enter an integer : '))
    if guess == number:
        print('Congratulations, you guessed it.')
        running = False
    elif guess < number:
        print('No, it is a little higher than that.')
    else:
        print('No, it is a little lower than that.')
else:
    print('The while loop is over.')  # 在这里你可以做你想做的任何事

print('Done')

# for

for i in range(1, 5): 
    print(i)

# enumerate

seq = (1, 2, 3)
for c, i in enumerate(seq):
    print('key {0}: value {1}'.format(c, i))
